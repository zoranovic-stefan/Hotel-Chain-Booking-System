# Project Setup

## Verify Python

python3 --version

## Verify pip

python3 -m pip --version

OR

pip --version

## Create Virtual Environment

python3 -m venv .venv
source venv/bin/activate # On Linux/macOS
venv\Scripts\activate # On Windows 

## Install Requirements

python3 -m pip install -r requirements.txt

## Run Local Server

python3 app.py
